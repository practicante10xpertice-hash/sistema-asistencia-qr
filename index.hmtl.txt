function gs(accion, payload = {}) {
  return new Promise((resolve, reject) => {
    if (typeof google !== 'undefined' && google.script && google.script.run) {
      google.script.run
        .withSuccessHandler(resolve)
        .withFailureHandler(reject)
        .dispatch(accion, payload);
    } else {
      // ── MODO DEMO (sin Apps Script) ───────────────
      console.log('📡 GS Call:', accion, payload);
      resolve(mockBackend(accion, payload));
    }
  });
}

function mockBackend(accion, payload) {
  const hoy = new Date().toISOString().split('T')[0];
  switch(accion) {
    case 'login': {
      if (payload.email === 'admin@empresa.com' && payload.password === 'admin123')
        return { ok:true, token:'DEMO_TOKEN', usuario:{ id:'USR001', nombre:'Administrador', email:'admin@empresa.com', rol:'Administrador' } };
      if (payload.email === 'usuario@empresa.com' && payload.password === 'usuario123')

      return { ok:true, message:'Empleado creado', id:nuevo.ID };
    }
    case 'actualizarEmpleado': {
      const idx = MOCK_DB.personal.findIndex(p => p.ID === payload.id);
      if (idx >= 0) MOCK_DB.personal[idx] = { ...MOCK_DB.personal[idx], ...payload.datos, ID:payload.id };
      return { ok:true, message:'Empleado actualizado' };
    }
    case 'eliminarEmpleado': {
      const idx = MOCK_DB.personal.findIndex(p => p.ID === payload.id);
      if (idx >= 0) MOCK_DB.personal[idx].Activo = 'false';
      return { ok:true, message:'Empleado eliminado' };

    case 'getQRToken': return { ok:true, token:'DEMO_QR_TOKEN', url: window.location.href + '?qr=DEMO_QR_TOKEN' };

      if (!emp) return { ok:false, message:'Empleado no encontrado' };
      const ahora = new Date();
      const hora = ahora.toTimeString().substring(0,8);
      const yaReg = MOCK_DB.asistencias.find(a => a.EmpleadoID === emp.ID && a.Fecha === hoy);
      if (yaReg) return { ok:false, message:'Ya registraste asistencia hoy' };
      const [hE,mE] = emp.HoraEntrada.split(':').map(Number);

      if (payload && payload.empleadoId) data = data.filter(a => a.EmpleadoID === payload.empleadoId);

    }
    case 'generarPDF': return { ok:false, message:'PDF disponible con Google Apps Script conectado' };

// AUTENTICACIÓN
// ═══════════════════════════════════════════════
async function handleLogin() {
  const email = document.getElementById('loginEmail').value.trim();
  const pass  = document.getElementById('loginPass').value;
  if (!email || !pass) { showToast('Ingresa email y contraseña', 'warning'); return; }
  const btn  = document.getElementById('loginBtnText');
  btn.innerHTML = '<span class="loader"></span>';
  try {
    const res = await gs('login', { email, password: pass });
    if (res.ok) {
      State.session = res.usuario;
      localStorage.setItem('asistenciaSession', JSON.stringify(res.usuario));
      showApp();
    } else {
      showToast(res.message || 'Error de autenticación', 'error');
    }
  } catch(e) {
    showToast('Error de conexión: ' + e, 'error');
  } finally {
    btn.innerHTML = 'Iniciar Sesión →';
  }
}

function logout() {
  localStorage.removeItem('asistenciaSession');
  State.session = null;
  document.getElementById('appPage').style.display  = 'none';
  document.getElementById('loginPage').style.display = 'flex';
  showToast('Sesión cerrada', 'info');
}

  document.getElementById('loginPage').style.display = 'none';
  document.getElementById('appPage').style.display   = 'flex';
  document.getElementById('sidebarName').textContent  = State.session.nombre;
  document.getElementById('sidebarRol').textContent   = State.session.rol;
  document.getElementById('sidebarAvatar').textContent = State.session.nombre[0].toUpperCase();
  updateTopbarDate();
  navigate('dashboard');
}

function editarEmpleado(id) {

  document.getElementById('emp_dni').value    = emp.DNI||emp['DNI']||'';
  document.getElementById('emp_cargo').value  = emp.Cargo||emp['Cargo']||'';
  document.getElementById('emp_area').value   = emp.Area||emp['Area']||'';
  document.getElementById('emp_horaEntrada').value    = emp.HoraEntrada||emp['HoraEntrada']||'08:00';
  document.getElementById('emp_horaSalida').value     = emp.HoraSalida||emp['HoraSalida']||'17:00';
  document.getElementById('emp_salidaAlmuerzo').value = emp.SalidaAlmuerzo||emp['SalidaAlmuerzo']||'';
  document.getElementById('emp_entradaTarde').value   = emp.EntradaTarde||emp['EntradaTarde']||'';
  document.getElementById('emp_salidaTarde').value    = emp.SalidaTarde||emp['SalidaTarde']||'';
  document.getElementById('emp_dobleTurno').checked   = (emp.DobleTurno||emp['DobleTurno']) === 'true';
  const diasLab = (emp.DiasLaborales||emp['DiasLaborales']||'').split(',').map(d => d.trim());
  document.querySelectorAll('.dia-btn').forEach(d => d.classList.toggle('selected', diasLab.includes(d.dataset.dia)));

async function guardarEmpleado() {
  const nombre = document.getElementById('emp_nombre').value.trim();
  const dni    = document.getElementById('emp_dni').value.trim();
  if (!nombre || !dni) { showToast('Nombre y DNI son obligatorios', 'warning'); return; }
  const diasSeleccionados = [...document.querySelectorAll('.dia-btn.selected')].map(d => d.dataset.dia);
  if (diasSeleccionados.length === 0) { showToast('Selecciona al menos un día laboral', 'warning'); return; }
  const datos = {
    nombre, dni,
    cargo:         document.getElementById('emp_cargo').value.trim(),
    area:          document.getElementById('emp_area').value.trim(),
    horaEntrada:   document.getElementById('emp_horaEntrada').value,
    horaSalida:    document.getElementById('emp_horaSalida').value,
    salidaAlmuerzo:document.getElementById('emp_salidaAlmuerzo').value,
    entradaTarde:  document.getElementById('emp_entradaTarde').value,
    salidaTarde:   document.getElementById('emp_salidaTarde').value,
    dobleTurno:    document.getElementById('emp_dobleTurno').checked,
    diasLaborales: diasSeleccionados
  };
  const btn = document.getElementById('btnGuardarEmpleado');
  btn.innerHTML = '<span class="loader"></span> Guardando...';
  btn.disabled  = true;
  let res;
  if (State.editingId) {
    res = await gs('actualizarEmpleado', { id: State.editingId, datos });
  } else {
    res = await gs('crearEmpleado', datos);
  }
  btn.disabled = false;
  if (res.ok) {
    showToast(res.message, 'success');
    closeModal('modalEmpleado');
    renderPersonal();
  } else {
    showToast(res.message || 'Error al guardar', 'error');
    btn.innerHTML = '💾 Guardar Empleado';
  }
}

function confirmarEliminar(id, nombre) {
  if (!confirm(`¿Eliminar a ${nombre}?\nEsta acción no se puede deshacer.`)) return;
  gs('eliminarEmpleado', { id }).then(res => {
    showToast(res.ok ? 'Empleado eliminado' : res.message, res.ok ? 'success' : 'error');
    if (res.ok) renderPersonal();
  });
}

async function renderAsistencia() {
  const c = document.getElementById('mainContent');
  c.innerHTML = '<div class="page-loader"><div class="spinner"></div><span>Generando QR...</span></div>';
  const [resQR, resPer] = await Promise.all([gs('getQRToken'), gs('getPersonal')]);
  if (resPer.ok) State.personal = resPer.data;

  c.innerHTML = `
  <div style="display:grid;grid-template-columns:1fr 1fr;gap:20px">
    <div class="section-card">
      <div class="section-head"><div class="section-title">📱 Código QR del Sistema</div></div>
      <div class="qr-container">
        <div class="qr-box"><div id="qrcode"></div></div>
        <div class="qr-info">
          <h3>Escanea para registrar asistencia</h3>
          <p>Los empleados deben escanear este código con su celular. El QR se vincula automáticamente a su perfil.</p>
        </div>
        <div style="display:flex;gap:10px;flex-wrap:wrap;justify-content:center">
          <button class="btn btn-outline" onclick="regenerarQR()">🔄 Regenerar QR</button>
          <button class="btn btn-primary" onclick="abrirModalScan()">✅ Registrar Manualmente</button>

          <thead><tr><th>Empleado</th><th>Hora</th><th>Estado</th></tr></thead>
          <tbody id="hoyTable"></tbody>
        </table>
      </div>
    </div>
  </div>`;

  // Generar QR
  const token = resQR.ok ? resQR.token : 'DEMO_TOKEN';
  State.qrToken = token;
  const qrUrl = resQR.ok ? resQR.url : (window.location.href + '?qr=' + token);
  new QRCode(document.getElementById('qrcode'), {
    text: qrUrl, width: 220, height: 220,
    colorDark:'#0f172a', colorLight:'#ffffff',
    correctLevel: QRCode.CorrectLevel.H
  });

  // Cargar asistencias de hoy
  const hoy = new Date().toISOString().split('T')[0];
  const resA = await gs('getAsistencias', { fechaInicio: hoy, fechaFin: hoy });
  const tb = document.getElementById('hoyTable');
  if (!resA.ok || resA.data.length === 0) {
    tb.innerHTML = '<tr><td colspan="3" style="text-align:center;color:var(--txt3);padding:24px">Sin marcaciones hoy</td></tr>';
  } else {
    resA.data.forEach(a => {
      tb.innerHTML += `<tr>
        <td><div class="cell-name">${a.NombreEmpleado||a['NombreEmpleado']}</div><div class="cell-sub">${a.Area}</div></td>
        <td><b>${(a.Hora||'').substring(0,5)}</b></td>
        <td><span class="badge badge-${(a.Estado||'').toLowerCase()}">${a.Estado}</span></td>
      </tr>`;
    });
  }
}

async function regenerarQR() {
  if (!confirm('¿Regenerar el QR? El código anterior dejará de funcionar.')) return;
  const res = await gs('regenerarQR');
  if (res.ok) { showToast('QR regenerado', 'success'); renderAsistencia(); }
  else showToast(res.message, 'error');
}

async function abrirModalScan() {
  const res = await gs('getPersonal');
  if (res.ok) State.personal = res.data;
  const sel = document.getElementById('scanEmpleadoSelect');
  sel.innerHTML = '<option value="">-- Selecciona un empleado --</option>';
  State.personal.forEach(emp => {
    sel.innerHTML += `<option value="${emp.ID||emp['ID']}">${emp.Nombre||emp['Nombre']} (${emp.DNI||emp['DNI']})</option>`;
  });
  openModal('modalQRScan');
}

async function registrarAsistenciaManual() {
  const empleadoId = document.getElementById('scanEmpleadoSelect').value;
  if (!empleadoId) { showToast('Selecciona un empleado', 'warning'); return; }
  const res = await gs('registrarQR', { token: State.qrToken || 'DEMO_QR_TOKEN', empleadoId });
  if (res.ok) {
    showToast(`✅ ${res.data.nombre} — ${res.data.estado} a las ${res.data.hora.substring(0,5)}`, 'success');
    closeModal('modalQRScan');
    if (State.currentPage === 'asistencia') renderAsistencia();
  } else {
    showToast(res.message, 'error');
  }
}

  btn.disabled = false;

function showToast(msg, type='info') {
  const icons = { success:'✅', error:'❌', warning:'⚠️', info:'ℹ️' };
  const toast = document.createElement('div');
  toast.className = `toast ${type}`;
  toast.innerHTML = `<span class="toast-icon">${icons[type]||'ℹ️'}</span><span>${msg}</span>`;
  document.getElementById('toastContainer').appendChild(toast);
  setTimeout(() => { toast.style.animation='toastIn .3s ease reverse'; setTimeout(()=>toast.remove(),300) }, 3800);
}

(function init() {
  // Revisar sesión guardada
  const saved = localStorage.getItem('asistenciaSession');
  if (saved) {
    try {
      State.session = JSON.parse(saved);
      showApp();
      return;
    } catch(e) { localStorage.removeItem('asistenciaSession'); }
  }
  // Revisar si viene de QR
  const params = new URLSearchParams(window.location.search);
  const qrParam = params.get('qr');
  if (qrParam) {
